\textbf{Lista de Exercícios}

\begin{enumerate}
    \item Enuncie os Axiomas de Peano

    \begin{equation}
        \alpha\beta=\beta\alpha
    \end{equation}
    para quaisquer inteiros $\alpha$ e $\beta$.
    
    \item Mostre como resolver o problema de encontrar um número real $x$ tal que:
    \begin{equation}
        x^2 - 5x + 6 = 0
    \end{equation}
\end{enumerate}
